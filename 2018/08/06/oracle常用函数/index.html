Oracle常用函数

编程 
/*
decode,将符合条件的数据，对应相应的值
*/
select decode(T_GRA,
720,'5',
420,'8',
120,'3',
'0'
) 
grades from student;
select t_num,t_nam,nvl(t_gra,0) from student;--若成绩为空，则将成绩视为0分
select '日期', trunc(sysdate,'month') from dual;--获取本月的第一天
select last_day(sysdate) from dual;
select next_day(sysdate,'星期一')from dual;--下个星期一的日期
select sysdate,round(sysdate) from dual;--前者显示年月日时分秒，后者只显示年月日
select to_date(to_char(sysdate,'yyyy-mm')||'-01','yyyy-mm-dd') from dual;--强制加-01，获取本月第一天
select trunc(3.14) from dual;--默认保留整数位
select trunc(3.14159267,5) from dual;--保留五位小数
select round(3.14159267,5) from dual;--保留五位小数
select round(sysdate,'month') from dual;--01-2月 -18,按日期四舍五入
select trunc(sysdate,'month') from dual;--01-1月 -18，按日期舍弃
select add_months(sysdate,-2) as dat from dual;--在sysdate基础上减少一个月

select trim('  123   ') from dual;--去除空格
select trim('a' from 'aabsasda') from dual;--去除指定的字符串
select round(3.14159267,2) from dual;--保留两位小数
select trunc(3.14159267,2) from dual;--保留两位
select round(3.14159267) from dual;--没有参数表示不会保留
select trunc(3.14159267) from dual;--没有参数表示不会保留
select trunc(sysdate-to_date('1994-9-20','yyyy-mm-dd')) from dual;--获取从1994年9月20日到今天的天数
select trunc((sysdate-to_date('1994-9-20','yyyy-mm-dd'))/7) from dual;--获取从1994年9月20日到今天的星期数
select trunc(months_between(sysdate,to_date('1994-9-20','yyyy-mm-dd'))) from dual;--获取从1994年9月20日到今天的月数
select to_number('123') from dual;--字符串转数值
select '123'+0 from dual;--直接在字符串后加0，实现转换数值
select to_date('1994-9-20','yyyy-mm-dd') from dual;--字符串转换日期
select replace('aaaaaasssssssssdfsaaaaaasd','a','0') from dual;--将源字符串中的a，替换为0
